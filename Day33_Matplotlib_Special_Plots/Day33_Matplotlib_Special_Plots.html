import matplotlib.pyplot as plt

categories = ['A', 'B', 'C', 'D']
values1 = [30, 40, 20, 50]
values2 = [20, 30, 25, 15]

fig, ax = plt.subplots(figsize=(8,6))
ax.bar(categories, values1, label='Group 1', color='skyblue', edgecolor='yellow')
ax.bar(categories, values2, bottom=values1, label='Group 2', color='salmon', edgecolor='black')
ax.set_title('Stacked Bar Plot', fontsize=20, fontweight='bold')
ax.legend()

plt.show()
