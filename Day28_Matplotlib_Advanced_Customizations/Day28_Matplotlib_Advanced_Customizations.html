1. Creating Subplots with plt.subplots()

In[2]:
import matplotlib.pyplot as plt
In[3]:
import numpy as np
In[4]:
x = np.linspace(0, 10, 100)
y1 = np.sin(x)
y2 = np.cos(x)

fig, axes = plt.subplots(1, 2, figsize=(10,4)) 

axes[0].plot(x, y1, 'r-')
axes[0].set_title('Sine Wave')

axes[1].plot(x, y2, 'b-')
axes[1].set_title('Cosine Wave')

plt.tight_layout()  
plt.show()

2. Shared Axes

In[5]:
fig, axes = plt.subplots(2, 1, sharex=True)

axes[0].plot(x, y1)
axes[0].set_title('Sine')

axes[1].plot(x, y2)
axes[1].set_title('Cosine')

plt.show()

3. Adding Annotations

In[10]:
fig, ax = plt.subplots()

ax.plot(x, y1)
ax.annotate('Max Point', xy=(np.pi/2, 1), xytext=(4, 1.5),
            arrowprops=dict(facecolor='black', shrink=0.05))
plt.show()
