import matplotlib.pyplot as plt
In [5]:
categories = ['MATHMATICS', 'ENGLISH','SCIENCE','HINDI','SOCIAL SCIENCE']
scores = [75,89,79,80,77]

plt.bar(categories, scores)
plt.title("Student Scores")
plt.xlabel("Subjects")
plt.ylabel("Scores")
plt.show()
