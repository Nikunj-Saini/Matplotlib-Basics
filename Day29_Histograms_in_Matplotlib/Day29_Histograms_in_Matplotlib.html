Multiple histograms

In [10]:
data1 = np.random.randn(1000)
data2 = np.random.randn(1000) + 2  # shifted mean

plt.hist(data1, bins=20, color='skyblue', alpha=0.7, label='Data 1')
plt.hist(data2, bins=20, color='salmon', alpha=0.7, label='Data 2')

plt.title("Overlapping Histograms", fontsize=16, fontweight='bold')
plt.xlabel("Value", fontsize=12)
plt.ylabel("Frequency", fontsize=12)
plt.legend()
plt.grid(axis='y', linestyle='--', alpha=0.7)

plt.show()
